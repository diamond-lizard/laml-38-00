   (body

     (p (list (kbd "This") "is" "the" "third"

     (p (list (kbd "This") "is" "the" "fourth" 
              'class "third" "paragraph" _ "."))  ;  

     (p (list "This" 
              (list (b "is") "the")
              "fifth" "paragraph" _ "."))  ;  

     (p (list "This"
               (list (b "is") "the" 'class "third")
              "sixth" "paragraph" _ "."))  ;  

     (p (list "This" "is" "t") _
        "h" _
        (list "e" "seventh")
        (list (list (list "paragraph"))) _ ".") ;  
     
   )
  )
