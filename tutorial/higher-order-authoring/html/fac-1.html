(load (string-append laml-dir "laml.scm"))
(laml-style "simple-html4.01-transitional-validating")

  
(define (fac n)
  (if (= 0 n) 1 (* n (fac (- n 1)))))

(write-html '(raw)
 (html 
  (head 
    (title "Factorials"))
  (body
    (h1 "Factorials")

    (table 'border "1"
     (tbody
      (tr 
        (td "1") (td (as-string (fac 1))))
      (tr 
        (td "2") (td (as-string (fac 2))))
      (tr 
        (td "3") (td (as-string (fac 3))))
      (tr 
        (td "4") (td (as-string (fac 4))))))
 ))
)

(end-laml)
